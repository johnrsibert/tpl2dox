352 void model_parameters::calculate_biomass_and_predicted_catch(void)
353 {
354  // calculate the biomass and predicted catch
355  if (!active(log_binit))
356  {
357  log_binit=log(obs_catch(1)/(q*effort(1)));
358  }
359  binit=exp(log_binit);
360  biomass[1]=binit;
361  if (active(k_devs))
362  {
363  k(1)=binit/beta;
364  for (int i=2;i<=nobs;i++)
365  {
366  k(i)=k(i-1)*exp(k_devs(i));
367  }
368  }
369  else
370  {
371  // set the whole vector equal to the constant k value
372  k=binit/beta;
373  }
374  // only calculate these for the standard deviation report
375  if (sd_phase())
376  {
377  k_1=k(1);
378  k_last=k(nobs);
379  k_ratio=k(nobs)/k(1);
380  k_change=k(nobs)-k(1);
381  }
382  // two time steps per year desired
383  int nsteps=2;
384  double delta=1./nsteps;
385  // Integrate the logistic dynamics over n time steps per year
386  for (int i=1; i<=nobs; i++)
387  {
388  bio_tmp=1.e-20+biomass[i];
389  c_tmp=0.;
390  for (int j=1; j<=nsteps; j++)
391  {
392  //This is the new biomass after time step delta
393  bio_tmp=bio_tmp*(1.+r*delta)/
394  (1.+ (r*pow(bio_tmp/k(i),m-1.)+f(i))*delta );
395  // This is the catch over time step delta
396  c_tmp+=f(i)*delta*bio_tmp;
397  }
398  pred_catch[i]=c_tmp; // This is the catch for this year
399  if (i<nobs)
400  {
401  biomass[i+1]=bio_tmp;// This is the biomass at the begining of the next
402  } // year
403  else
404  {
405  B_projected=bio_tmp; // get the projected biomass for std dev report
406  }
407  }
408 
409 }

411 void model_parameters::calculate_the_objective_function(void)
412 {
413  if (!active(effort_devs))
414  {
415  ff=nobs/2.*log(norm2(log(obs_catch)-log(1.e-10+pred_catch)));
416  }
417  else if(!active(k_devs))
418  {
419  ff= .5*(size_count(obs_catch)+size_count(effort_devs))*
420  log(norm2(log(obs_catch)-log(1.e-10+pred_catch))
421  +0.1*norm2(effort_devs));
422  }
423  else
424  {
425  ff= .5*( size_count(obs_catch)+size_count(effort_devs)
426  +size_count(k_devs) )*
427  log(norm2(log(obs_catch)-log(pred_catch))
428  + 0.1*norm2(effort_devs)+10.*norm2(k_devs));
429  }
430  // Bayesian contribution for Pella Tomlinson m
431  ff+=2.*square(log(m-1.));
432  if (initial_params::current_phase<3)
433  {
434  ff+=1000.*square(log(mean(f)/.4));
435  }
