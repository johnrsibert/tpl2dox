337  /**
338  \ingroup PROCEDURE_SECTION
339  Function to compute mortality and survival rate. */
340 void model_parameters::get_mortality_and_survivial_rates(void)
341 {
342  int i, j;
343  // calculate the selectivity from the sel_coffs
344  for (j=1;j<nages;j++)
345  {
346  log_sel(j)=log_sel_coff(j);
347  }
348  // the selectivity is the same for the last two age classes
349  log_sel(nages)=log_sel_coff(nages-1);
350 
351  // This is the same as F(i,j)=exp(q)*effert(i)*exp(log_sel(j));
352  F=outer_prod(mfexp(log_q)*effort,mfexp(log_sel));
353  if (active(effort_devs))
354  {
355  for (i=1;i<=nyrs;i++)
356  {
357  F(i)=F(i)*exp(effort_devs(i));
358  }
359  }
360  // get the total mortality
361  Z=F+M;
362  // get the survival rate
363  S=mfexp(-1.0*Z);
364 
365 }
366 // end of function void model_parameters::get_mortality_and_survivial_rates(void)

368  /**
369  \ingroup PROCEDURE_SECTION
370  Compute numbers at age. */
371 void model_parameters::get_numbers_at_age(void)
372 {
373  int i, j;
374  log_initpop=log_relpop+log_popscale;
375  for (i=1;i<=nyrs;i++)
376  {
377  N(i,1)=mfexp(log_initpop(i));
378  }
379  for (j=2;j<=nages;j++)
380  {
381  N(1,j)=mfexp(log_initpop(nyrs+j-1));
382  }
383  for (i=1;i<nyrs;i++)
384  {
385  for (j=1;j<nages;j++)
386  {
387  N(i+1,j+1)=N(i,j)*S(i,j);
388  }
389  }
390  // calculated predicted numbers at age for next year
391  for (j=1;j<nages;j++)
392  {
393  predicted_N(j+1)=N(nyrs,j)*S(nyrs,j);
394  ratio_N(j+1)=predicted_N(j+1)/N(1,j+1);
395  }
396  // calculated predicted Biomass for next year for
397  // adjusted profile likelihood
398  pred_B=(predicted_N * relwt);
399 
400 }
401 // end of function void model_parameters::get_numbers_at_age(void)

406 void model_parameters::get_catch_at_age(void)

408  C=elem_prod(elem_div(F,Z),elem_prod(1.-S,N));

413  /**
414  \ingroup PROCEDURE_SECTION
415  Compute objective function. */
416 void model_parameters::evaluate_the_objective_function(void)
417 {
418  // penalty functions to ``regularize '' the solution
419  f+=.01*norm2(log_relpop);
420  avg_F=sum(F)/double(size_count(F));
421  if (last_phase())
422  {
423  // a very small penalty on the average fishing mortality
424  f+= .001*square(log(avg_F/.2));
425  }
426  else
427  {
428  f+= 1000.*square(log(avg_F/.2));
429  }
430  f+=0.5*double(size_count(C)+size_count(effort_devs))
431  * log( sum(elem_div(square(C-obs_catch_at_age),.01+C))
432  + 0.1*norm2(effort_devs));
433 
434 }
435 // end of function void model_parameters::evaluate_the_objective_function(void)
